694 node* parse_expression( char expression[]){
695 
696  char token[2000][30];
697  for(int k=0;k<1000;k++){
698  for(int w=0;w<30;w++){
699  token[k][w]='\0';
700  }
701  }
702  node** stack_operators;
703  node** stack_operands;
704  stack_operators= new node* [1000];
705  stack_operands= new node* [1000];
706  int count_operands=0;
707  int count_operators=0;
708  unsigned int i=0;
709  bool end_symbol;
710  int count_token=0;
711  node *result=NULL;
712  while(i<strlen(expression)){
713  int j=0;
714  end_symbol=false;
715  if(expression[i]==' ')
716  i++;
717  else{
718  while(!end_symbol){
719  //char symbol[30];
720  for(int k=0;k<30;k++)
721  //symbol[k]='\0';
722  if(expression[i]=='(' || expression[i]==')' || expression[i]=='-' || expression[i]=='+' || expression[i]=='*' || expression[i]=='/'){
723  node *el=new node;
724  if(expression[i]=='('){
725  el->root=symbols[0];
726  }
727  if(expression[i]==')'){
728  el->root=symbols[1];
729  }
730  if(expression[i]=='+'){
731  el->root=symbols[2];
732  }
733  if(expression[i]=='-'){
734  el->root=symbols[3];
735  }
736  if(expression[i]=='*'){
737  el->root=symbols[4];
738  }
739  if(expression[i]=='/'){
740  el->root=symbols[5];
741  }
742 
743 
744 
745  if(count_operators>0 && expression[i]!=')'){
746 
747  node *x=new node;
748  copy_operator(stack_operators[count_operators-1], (node **)&x,NULL);
749  x->children=NULL;
750 
751  if(x->root->precedence < el->root->precedence || el->root==symbols[0]){
752 
753  stack_operators[count_operators]=el;
754  stack_operators[count_operators]->children=NULL;
755 
756  count_operators++;
757  delete_individual(x);
758 
759  }
760 
761 
762  else{
763  delete_individual(stack_operators[count_operators-1]);
764  count_operators--;
765 
766  int num_children=0;
767  (x)->children=new node* [(x)->root->arity];
768 
769  while(num_children<x->root->arity){
770 
771  node *op=new node;
772 
773 
774  copy_operand(stack_operands[count_operands-1], (node **)&op,NULL);
775 
776  delete_individual(stack_operands[count_operands-1]);
777  count_operands--;
778  (x)->children[x->root->arity-1-num_children]=op;
779  (x)->children[x->root->arity-1-num_children]->parent=x;
780  num_children++;
781  }
782 
783  stack_operands[count_operands]=x;
784  count_operands++;
785  el->children=NULL;
786  stack_operators[count_operators]=el;
787  count_operators++;
788  }
789  }
790  if(expression[i]==')'){
791  bool open_par=false;
792  do{
793  node *oper=new node;
794  copy_operator(stack_operators[count_operators-1], (node **)&oper,NULL);
795  oper->children=NULL;
796  delete_individual(stack_operators[count_operators-1]);
797  count_operators--;
798  int num_children=0;
799  oper->children=new node* [oper->root->arity];
800  if(oper->root!=symbols[0]){
801  while(num_children<oper->root->arity){
802  node *op=new node;
803  copy_operand(stack_operands[count_operands-1], (node **)&op,NULL);
804  delete_individual(stack_operands[count_operands-1]);
805  count_operands--;
806  oper->children[oper->root->arity-1-num_children]=op;
807  oper->children[oper->root->arity-1-num_children]->parent=oper;
808  num_children++;
809  }
810  stack_operands[count_operands]=oper;
811  count_operands++;
812  }
813  else{
814  open_par=true;
815  delete_individual(oper);
816  }
817  }while(!open_par);
818  el->children=NULL;
819  delete_individual(el);
820  }
821  if(count_operators==0 && expression[i]!=')'){
822 
823  el->children=NULL;
824  stack_operators[count_operators]=el;
825  count_operators++;
826 
827  }
828  i++;
829  }
830  else{
831 
832  while(expression[i]!=' '){
833  token[count_token][j]=expression[i];
834  j++;
835  i++;
836  }
837  if(strlen(token[count_token])>0){
838  node *el=new node;
839  el->root=symbols[search(token[count_token])];
840  el->children=NULL;
841  stack_operands[count_operands]=el;
842  count_operands++;
843  count_token++;
844  }
845  }
846  end_symbol=true;
847  }
848  }
849  }
850 
851  if(count_operands>0){
852  result=new node;
853  copy_operand(stack_operands[count_operands-1], (node **)&result, NULL);
854  delete_individual(stack_operands[count_operands-1]);
855  count_operands--;
856  }
857  delete[] stack_operators;
858  delete[] stack_operands;
859 
860  return result;
861 }

1163  }
